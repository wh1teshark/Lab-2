        System.out.println("–ü—Ä–µ–æ–±—Ä–∞–∑–æ–≤–∞–Ω–Ω–∞—è –º–∞—Ç—Ä–∏—Ü–∞:");
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < maxLength; j++) {
                if (j < matrix[i].length) {

    private static int getMaxRowLength(int[][] matrix) {
        int maxLength = 0;
        for (int[] row : matrix) {
            if (row.length > maxLength) {
                maxLength = row.length;
            }
        }
        return maxLength;
